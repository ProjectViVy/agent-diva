```rust
#[tauri::command]
async fn get_sandbox_config(state: tauri::State<'_, AppState>) -> Result<serde_json::Value, String> {
    let config = state.config.read().await;
    let sandbox = config.sandbox.clone();
    serde_json::to_value(&sandbox).map_err(|e| e.to_string())
}

#[tauri::command]
async fn save_sandbox_config(state: tauri::State<'_, AppState>, config: serde_json::Value) -> Result<(), String> {
    let sandbox_config: agent_diva_core::config::SandboxConfig = serde_json::from_value(config)
        .map_err(|e| format!("Invalid sandbox config: {}", e))?;
    let mut cfg = state.config.write().await;
    cfg.sandbox = sandbox_config;
    // Save config to disk
    let config_path = agent_diva_core::config::default_config_path()
        .map_err(|e| e.to_string())?;
    let yaml = serde_yaml::to_string(&*cfg).map_err(|e| e.to_string())?;
    std::fs::write(&config_path, yaml).map_err(|e| e.to_string())?;
    Ok(())
}
```
